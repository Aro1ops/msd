### 3. Изменение JavaScript

Теперь реализуем логику для новых функций, таких как профиль, промокоды и режим разработчика.

```javascript
document.addEventListener("DOMContentLoaded", () => {
    const coinImage = document.getElementById("coin-image");
    const clickCounter = document.getElementById("click-counter");
    const energyFill = document.getElementById("energy-fill");
    const energyLabel = document.getElementById("energy-label");
    const changeButton = document.getElementById("change-button");
    const storeButton = document.getElementById("store-button");
    const profileButton = document.getElementById("profile-button");
    const changeMenu = document.getElementById("change-menu");
    const storeMenu = document.getElementById("store-menu");
    const profileMenu = document.getElementById("profile-menu");
    const promoCodeMenu = document.getElementById("promo-code-menu");
    const developerMode = document.getElementById("developer-mode");
    const backButton = document.getElementById("back-button");
    const backStoreButton = document.getElementById("back-store-button");
    const backProfileButton = document.getElementById("back-profile-button");
    const backPromoButton = document.getElementById("back-promo-button");
    const backDevButton = document.getElementById("back-dev-button");
    const upgradeClickButton = document.getElementById("upgrade-click");
    const upgradeEnergyButton = document.getElementById("upgrade-energy");
    const reduceRegenButton = document.getElementById("reduce-regeneration");
    const nicknameDisplay = document.getElementById("nickname");
    const promoInput = document.getElementById("promo-input");
    const checkPromoButton = document.getElementById("check-promo-button");
    const promoMessage = document.getElementById("promo-message");
    const savePromoCodeButton = document.getElementById("save-promo-code");
    const coinsAmountInput = document.getElementById("coins-amount");
    const giveCoinsButton = document.getElementById("give-coins");
    const maxEnergyButton = document.getElementById("max-energy");
    const resetAllButton = document.getElementById("reset-all");
    const playerList = document.getElementById("player-list");

    let clickCount = 0;
    let energy = 2000;
    let maxEnergy = 2000;
    let clickValue = 1;
    let regenTime = 10000;
    const clickCounts = JSON.parse(localStorage.getItem("clickCounts")) || {
        coin1: 0,
        coin2: 0,
        coin3: 0,
        coin4: 0
    };
    const clickUpgrades = JSON.parse(localStorage.getItem("clickUpgrades")) || {
        clickPower: 0,
        energyUpgrade: 0,
        reduceRegeneration: 0
    };

    function updateEnergy() {
        energyFill.style.width = (energy / maxEnergy * 100) + "%";
        energyLabel.textContent = `${energy} / ${maxEnergy}`;
    }

    function updateClickCounter() {
        clickCounter.textContent = clickCount;
    }

    coinImage.addEventListener("click", () => {
        if (energy > 0) {
            clickCount += clickValue;
            energy -= 1;
            updateClickCounter();
            updateEnergy();
            animateClick();
        }
    });

    function animateClick() {
        const animation = document.createElement("div");
        animation.textContent = `+${clickValue}`;
        animation.className = "click-animation";
        document.body.appendChild(animation);
        animation.style.left = `${event.clientX}px`;
        animation.style.top = `${event.clientY}px`;
        setTimeout(() => animation.remove(), 1000);
    }

    changeButton.addEventListener("click", () => {
        changeMenu.classList.remove("hidden");
        document.body.style.pointerEvents = "none";
    });

    backButton.addEventListener("click", () => {
        changeMenu.classList.add("hidden");
        document.body.style.pointerEvents = "auto";
    });

    storeButton.addEventListener("click", () => {
        storeMenu.classList.remove("hidden");
        document.body.style.pointerEvents = "none";
    });

    backStoreButton.addEventListener("click", () => {
        storeMenu.classList.add("hidden");
        document.body.style.pointerEvents = "auto";
    });

    profileButton.addEventListener("click", () => {
        profileMenu.classList.remove("hidden");
        document.body.style.pointerEvents = "none";
        nicknameDisplay.textContent = "Игрок"; // пример имени игрока
        updateUpgradesList();
    });

    backProfileButton.addEventListener("click", () => {
        profileMenu.classList.add("hidden");
        document.body.style.pointerEvents = "auto";
    });

    checkPromoButton.addEventListener("click", () => {
        const promoCode = promoInput.value;
        if (promoCode === "FREECOINS") {
            clickCount += 100;
            promoMessage.textContent = "Промокод активирован! +100 монет";
            updateClickCounter();
        } else {
            promoMessage.textContent = "Неверный промокод.";
        }
    });

    backPromoButton.addEventListener("click", () => {
        promoCodeMenu.classList.add("hidden");
        document.body.style.pointerEvents = "auto";
    });

    // Установка обработчиков для режима разработчика
    savePromoCodeButton.addEventListener("click", () => {
        const promoCode = document.getElementById("promo-input-dev").value;
        const coinsAmount = parseInt(coinsAmountInput.value, 10);
        if (promoCode) {
            localStorage.setItem(`promo-${promoCode}`, coinsAmount);
            alert(`Промокод ${promoCode} добавлен с количеством монет: ${coinsAmount}`);
        }
    });

    giveCoinsButton.addEventListener("click", () => {
        const amount = parseInt(coinsAmountInput.value, 10);
        clickCount += amount;
        updateClickCounter();
    });

    maxEnergyButton.addEventListener("click", () => {
        energy = maxEnergy;
        updateEnergy();
    });

    resetAllButton.addEventListener("click", () => {
        localStorage.clear();
        clickCount = 0;
        energy = maxEnergy;
        updateClickCounter();
        updateEnergy();
        alert("Все данные сброшены.");
    });

    function updateUpgradesList() {
        const upgradesList = document.getElementById("upgrades-list");
        upgradesList.innerHTML = `
            <div>Прокачка кликов: ${clickUpgrades.clickPower}</div>
            <div>Прокачка энергии: ${clickUpgrades.energyUpgrade}</div>
            <div>Уменьшение регенерации: ${clickUpgrades.reduceRegeneration}</div>
        `;
    }
    
    // Загрузка сохраненных данных
    clickCount = parseInt(localStorage.getItem("clickCount")) || 0;
    energy = parseInt(localStorage.getItem("energy")) || maxEnergy;
    updateClickCounter();
    updateEnergy();
});
```
